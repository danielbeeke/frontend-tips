/**
 * Implements hook_preprocess_block().
 */
function vdh_jacket_preprocess_block(&$variables) {
  $blocks_with_wrapper = array(
    'vdh_basic-footer',
  );

  if (!in_array($variables['block']->bid, $blocks_with_wrapper)) {
    $variables['hide_wrappers'] = TRUE;
  }
}
